1.内置类型

1）整数

a = 123
print(a)
print(type(a))

2）列表

b = [1, 2, 3]
print(b)
print(type(b))

[1, 2, 3] 
<class 'list'>

3）字符串

string = 'string'
print(string)
print(type(string))
bs = b'bytes'

8.类

class student(object) :

    def __init__(self, id, name) :
        self.id = id
        self.__name = name

    def print_student(self) :
        print(self.id, self.__name)

a = student('01111295', 'Leo')
a.print_student()
try :
    print('id:', a.id)
    print('name:', a.__name)
except :
    print('error')

01111295 Leo 
id: 01111295 
error

9.继承

class fruit(object) :

    def __init__(self, name) :
        self.name = name

    def output_name(self) :
        print(self.name)


class apple(fruit) :

    def __init__(self, name) :
        self.name = name

a = apple("apple")
a.output_name()
type(a)

apple 
__main__.apple
